# EmployeesSQL Challenge 

Employees
-
emp_no PK int 
emp_title_id varchar(10) FK >- Titles.emp_title_id
birth_data date
first_name varchar(20)
last_name (20)
sex varchar(2) 
hiredate date

Titles
-
emp_title_id  PK varchar(10)
title  varchar(20)

Salaries as S
----
emp_no int FK >- Employees.emp_no
Salary money

Dept_emp as DE
----
emp_no  FK >- Employees.emp_no 
dept_no string FK >- Departments.dept_no

Departments as D
----
dept_no varchar(10) PK
dept_name varchar(20)

Dept_manager as DM
----
manager varchar(20) FK >- Departments.dept_no
emp_no int FK >- Employees.emp_no
